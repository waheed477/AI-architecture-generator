```jsx
// /components/ThreeDModel.js - USE THIS EXACT CODE
import { Canvas } from "@react-three/fiber";
import { useState, useEffect } from "react";

export default function ThreeDModel({ layoutData }) {
  const [model, setModel] = useState(null);

  useEffect(() => {
    if (layoutData) {
      setModel(layoutData); // This would be the processed layout data
    }
  }, [layoutData]);

  return (
    <Canvas>
      {model && model.map((room, index) => (
        <mesh key={index} position={room.position}>
          <boxBufferGeometry args={[room.width, room.height, room.depth]} />
          <meshStandardMaterial color="orange" />
        </mesh>
      ))}
    </Canvas>
  );
}'
